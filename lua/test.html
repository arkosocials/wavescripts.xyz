local Pets = require(game:GetService("ReplicatedStorage").Library.Directory.Pets) local fromPet = "Blue Fish" local toPet = "Huge Cat" if fromPet == "" or toPet == "" then warn("Hey! Please Fill in Both 'From Pet' and 'To Pet' Names!") return end if Pets[fromPet] and Pets[toPet] then for i, v in pairs(Pets[fromPet]) do Pets[fromPet][i] = nil end for i, v in pairs(Pets[toPet]) do Pets[fromPet][i] = v end print("Successful! You Spawned", fromPet, "to", toPet) else warn("Oops! Invalid Pet Names! Please Confirm the Pets Names!") end
